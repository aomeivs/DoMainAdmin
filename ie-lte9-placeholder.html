htlm，ie-lte9.js 

    除登录、注册以及有“切换卡效果的form表单”页面，
    其它页面可以直接使用ie-lte9.js替换原ie8.js
    ---并调用以下方法。
javascript，ie-lte9.js 

    //1.引入ie-lte9.js
    //2.$.placeholderInit({replaceAll:false} //初始化插件，并根据参数是否执行替换
    //参数说明：replaceAll 是否替换页面所有input[placeholder]。
    //当页面支持placeholder时，返回true.
    //当设置replaceAll:false切页面不支持placeholder时返回false。

    $(function () {
            if(!$.placeholderInit({replaceAll:false})){
    //不全部替换，使用自定义逻辑：
                $('[placeholder]').each(function() {
                    var $elem   = $(this)
                    // 文本
                    var word = $elem.attr('placeholder')
                    $elem.placeholder({
    //                        word: word,
                        diffPaddingLeft:64
                    })
                })
            }
    })
